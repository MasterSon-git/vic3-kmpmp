kmpmp_player_boost_art_of_deal_decision = {
    is_shown = {
        NOT = {
            has_variable = kmpmp_player_boost_recently_adjusted
        }
    }

    possible = {
        always = yes
    }

    when_taken = {
        set_variable = {
            name = kmpmp_player_boost_recently_adjusted
            value = yes
            days = 7300 # 20 years
        }
        remove_player_boost_modifiers = yes
        add_modifier = {
            name = kmpmp_player_boost_art_of_deal
        }
    }

    ai_chance = {
        base = 0	
    }
}
