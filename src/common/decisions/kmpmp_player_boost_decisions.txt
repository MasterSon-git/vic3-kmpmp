kmpmp_player_boost_art_of_deal_decision = {
    is_shown = {
        always = yes
    }

    possible = {
        NOT = {
            has_modifier = kmpmp_player_boost_adjustment
        }
    }

    when_taken = {
        add_modifier = {
            name = kmpmp_player_boost_adjustment
            years = 15
        }
        remove_player_boost_modifiers = yes
        add_modifier = {
            name = kmpmp_player_boost_art_of_deal
        }
    }

    ai_chance = {
        base = 0	
    }
}
