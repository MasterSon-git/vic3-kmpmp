kmpmp_emser_depeche = {
	is_shown = {
		OR = {
			c:NGF ?= owner
			c:PRU ?= owner
		}
		NOT = { exists = c:GER }
		NOT = { has_variable = took_decision_emser_depeche }
	}

	possible = {
		exists = c:NGF
		c:NGF ?= this
		has_technology_researched = nationalism
		year <= 1871
		has_war_with = c:FRA
	}
	when_taken = {
		add_loyalists = {
 			value = 0.08
 			culture = cu:north_german
		}
		add_modifier = {
			name = kmpmp_retake_whats_rightfully_german
			years = 5
		}
		set_variable = took_decision_emser_depeche
	}
}

kmpmp_austrian_hungarian_agreement = {
	is_shown = {
		NOT = { 
			has_variable = took_decision_austrian_hungarian_agreement
		}
		c:AUS ?= owner
		year >= 1845
		NOT = {
			country_has_primary_culture = cu:hungarian
		}
	}

	possible = {
		has_technology_researched = nationalism
	}

	when_taken = {
		set_variable = took_decision_austrian_hungarian_agreement

		if = {
			limit = {
				NOT = {
					has_law = law_type:law_racial_segregation
				}
			}
			activate_law = law_type:law_racial_segregation
		}

		add_modifier = {
			name = kmpmp_austrian_the_great_reform
			years = 5
			is_decaying = yes
		}		
	}
}

kmpmp_reino_unido_force_opponent_protectorate = {
	is_shown = {
		OR = {
			has_journal_entry = je_reino_unido_por
			has_journal_entry = je_reino_unido_brz
		}
		NOR = {
			c:POR ?= {
				is_direct_subject_of = c:BRZ
				is_diplomatic_play_enemy_of = c:BRZ
			}
			c:BRZ ?= {
				is_direct_subject_of = c:POR
				is_diplomatic_play_enemy_of = c:POR
			}
		}
	}

	possible = {
		aggressive_diplomatic_plays_permitted = yes
	}

	when_taken = {
		if = {
			limit = {
				c:BRZ ?= ROOT
			}
			create_diplomatic_play = {
				#name = 
				type = dp_make_protectorate
				target_country = c:POR
			}
		}
		else_if = {
			limit = {
				c:POR ?= ROOT
			}
			create_diplomatic_play = {
				#name = 
				type = dp_make_protectorate
				target_country = c:BRZ
			}
		}
	}
}
