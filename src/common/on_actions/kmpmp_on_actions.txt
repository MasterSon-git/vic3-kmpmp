on_monthly_pulse_country = {
    on_actions = {
        buff_nation_effect_for_unification
        player_boost_effect_state_unincorporated_education_access
    }
}

buff_nation_effect_for_unification = {
    effect = {
        if = {
            limit = {
                exists = c:ITA
                owner = c:ITA
            }
            if = {
                limit = {
                    NOT = {
                        has_modifier = kmpmp_buff_medium_nation
                    }
                }
                add_modifier = {
                    name = kmpmp_buff_medium_nation
                }
            }
            if = {
                limit = {
                    has_modifier = kmpmp_buff_small_nation
                }
                remove_modifier = kmpmp_buff_small_nation
            }
        }

        if = {
            limit = {
                exists = c:SCA
                owner = c:SCA
            }
            if = {
                limit = {
                    NOT = {
                        has_modifier = kmpmp_buff_medium_nation
                    }
                }
                add_modifier = {
                    name = kmpmp_buff_medium_nation
                }
            }
            if = {
                limit = {
                    has_modifier = kmpmp_buff_small_nation
                }
                remove_modifier = kmpmp_buff_small_nation
            }
        }
    }
}

player_boost_effect_state_unincorporated_education_access = {
    effect = {
        if = {
            limit = {
                has_modifier = kmpmp_player_boost_imperialism_oversee_empire
            }
            every_scope_state = {
                limit = {
                    is_incorporated = no
                }
                add_modifier = {
                    name = kmpmp_player_boost_imperialism_oversee_empire_state_unincorporated_education_access
                    days = 30
                }
            }
        }
    }
}
