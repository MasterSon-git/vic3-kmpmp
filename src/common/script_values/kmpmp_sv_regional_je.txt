kmpmp_sv_count_states_master_of_mediterranean_sea = {
    value = 0
    every_in_global_list = {
        variable = states_je_master_of_mediterranean_sea
        limit = {
            any_scope_state = { 
                owner = ROOT
                count = all
            }
            trigger_if = {
                limit = {
                    is_island = yes
                }
                any_scope_state = {
                    is_incorporated = yes
                }
            }
        }
        add = 1
    }
}
