	}

	country_trigger = {
		has_technology_researched = nationalism #political_agitation
	}

	interest_group_leader_trigger = {
		ideology_fascist_valid_trigger = yes
	}

	non_interest_group_leader_trigger = {
		# Agitator ideologies should disapprove of a current law
		NAND = {
			has_role = agitator
			owner = {
				has_law_or_variant = law_type:law_autocracy
				has_law_or_variant = law_type:law_ethnostate
				has_law_or_variant = law_type:law_outlawed_dissent
				has_law_or_variant = law_type:law_militarized_police
			}
		}
		interest_group = {
			ideology_fascist_valid_trigger = yes
		}
	}

	interest_group_leader_weight = {
		value = 50

		# More likely if people are angry
		add = {
			value = owner.radical_fraction
			multiply = 100
		}

		# More likely if issues are relevant
		if = {
			limit = {
				owner.ideology_fascist_disfavoured_laws > 0
			}
			add = {
				value = owner.ideology_fascist_disfavoured_laws
				multiply = 8
			}
		}

		# Less likely if issues are not relevant
		if = {
			limit = {
				owner.ideology_fascist_favoured_laws > 0
			}
			subtract = {
				value = owner.ideology_fascist_favoured_laws
				multiply = 8
			}
		}

		# More likely if paying war reparations
		if = {
			limit = {
				any_country = {
					any_scope_treaty = {
						is_enforced = yes
						any_scope_article = {
							has_type = money_transfer
							source_country = root.owner
						}
					}
				}
			}
			add = 25
		}
		# More likely with a convenient scapegoat
		if = {
			limit = {
				owner = {
					any_scope_state = {
						is_incorporated = yes
						any_scope_pop = {
							pop_acceptance < acceptance_status_4
							total_size >= 1000
						}
					}
				}
			}
			add = 25
		}
		# more likely if bankrupt
		if = {
			limit = {
				owner = {
					OR = {
						in_default = yes
						has_modifier = declared_bankruptcy
					}
				}
			}
			add = 25
		}
		# Less likely in council republic
		if = {
			limit = {
				owner = {
					has_law_or_variant = law_type:law_council_republic
				}
			}
			add = -75
		}
		# more likely if angry non-unions
		if = {
			limit = {
				ig_approval <= -5
				NOT = { is_interest_group_type = ig_trade_unions }
			}
			subtract = {
				value = ig_approval
				multiply = 5
			}
		}
		# less likely if content without ethnostate or Corporate State
		if = {
			limit = {
				owner = {
					NOR = {
						has_law_or_variant = law_type:law_ethnostate
						has_law_or_variant = law_type:law_corporate_state
					}
				}
				ig_approval >= 5
			}
			subtract = {
				value = ig_approval
				multiply = 5
			}
		}
		# more likely for PB and armed forces
		if = {
			limit = {
				OR = {
					is_interest_group_type = ig_petty_bourgeoisie
					is_interest_group_type = ig_armed_forces
				}
			}
			add = 50
		}
		# less likely for unions and intelligentsia
		if = {
			limit = {
				OR = {
					is_interest_group_type = ig_trade_unions
					is_interest_group_type = ig_intelligentsia
				}
			}
			add = -25
		}
		# more likely if in government with ethnostate or Corporate State
		if = {
			limit = {
				owner = {
					OR = {
						has_law_or_variant = law_type:law_ethnostate
						has_law_or_variant = law_type:law_corporate_state
					}
					NOR = {
						is_enacting_law = law_type:law_multicultural
						is_enacting_law = law_type:law_national_supremacy
						is_enacting_law = law_type:law_racial_segregation
						is_enacting_law = law_type:law_cultural_exclusion
					}
				}
				is_in_government = yes
			}
			add = 50
		}
		# conservative and/or reactionary leaders have a higher spawn chance if the Anti-Dreyfusards won.
		if = {
			limit = {
				has_variable = anti_dreyfusard_victory
			}
			add = 50
		}
		# red scare can make you more fascist
		if = {
			limit = {
				has_variable = fascist_uptick
			}
			multiply = 1.5
		}
		# Fascism JE causes an uptick in fascism
		if = {
			limit = {
				owner = {
					OR = {
						has_journal_entry = je_fascism_1
						has_variable = completed_fascism_1_var
					}
				}
			}
			multiply = 1.25
		}
		if = { #Brazil has Integralists instead
			limit = {
				owner = {
					c:BRZ ?= this
				}
			}
			multiply = 0
		}
		if = {
			limit = {
				OR = {
					has_variable = communist_ig_var
					has_variable = chose_vanguardism
					has_variable = chose_anarchism
					has_variable = chose_communism
					has_variable = chose_ethno_nationalism
				}
			}
			multiply = 0.1
		}
		# Minority cultures won't favor highly ethno-supremacist ideologies
		if = {
			limit = {
				leader = {
					culture = {
						NOT = { is_primary_culture_of = ROOT }
					}
				}
			}
			multiply = 0
		}
	}
