REPLACE:je_meiji_main = {
	icon = "gfx/interface/icons/event_icons/event_scales.dds"

	group = je_group_meiji_restoration

	scripted_button = je_iwakura_mission_button
	scripted_button = je_end_edo_system_button

	immediate = {
		set_variable = { name = meiji_var value = 0 }
	}

	complete = {
		OR = {
			AND = {
				scope:journal_entry = {
					is_goal_complete = yes
				}
				hidden_trigger = {
					has_variable = completed_je_meiji_army
					has_variable = completed_je_meiji_economy
					OR = {
						AND = {
							has_dlc_feature = ep2_content
							has_variable = iwakura_mission_finished
						}
						AND = {
							NOT = { has_dlc_feature = ep2_content }
							has_variable = completed_je_meiji_diplomacy
						}
					}
				}
			}
			custom_tooltip = {
				text = meiji_timeout_some_progress
				always = no
			}
		}
	}

	on_complete = {
		trigger_event = {
			id = meiji.2
		}
		# kmpmp - show that it needs to be completed before year 1885 otherwise you don't get the buff "Empire of the rising Sun"
        show_as_tooltip = {
            custom_tooltip = kmpmp_japan_buff_meiji_complete_condition_tt
            add_modifier = {
                name = kmpmp_japan_buff_meiji_complete
                years = 15
            }
        }
	}

	on_timeout = {
		hidden_effect = {
			if = {
				limit = {
					OR = {
						has_variable = completed_je_meiji_army
						has_variable = completed_je_meiji_economy
						has_variable = iwakura_mission_finished
						has_variable = completed_je_meiji_diplomacy
					}
				}
				trigger_event = { id = meiji.2 }
			}
			else = {
				trigger_event = { id = meiji.14 }
			}
		}
	}

	fail = {
		custom_tooltip = {
			text = meiji_timeout_no_progress
			always = no
		}
	}

	on_fail = {
		show_as_tooltip = {
			trigger_event = { id = meiji.14 }
		}
	}

	timeout = 4380

	on_monthly_pulse = {
		random_events = {
			100 = 0
			10 = meiji.4
			10 = meiji.5
			10 = meiji.6
		}
	}

	current_value = {
		value = var:meiji_var
	}

	goal_add_value = {
		add = 3
	}

	#progressbar = yes

	weight = 10000

	should_be_pinned_by_default_uninvolved_or_context = yes
}
