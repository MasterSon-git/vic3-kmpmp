je_sick_man_main = {
	icon = "gfx/interface/icons/event_icons/event_skull.dds"

	group = je_group_tanzimat

	is_shown_in_lobby = {
		c:TUR = THIS
    }

	complete = {
		scope:journal_entry = {
			is_goal_complete = yes
		}
	}

	on_complete = {
		remove_modifier = sick_man_of_europe
		trigger_event = {
			id = sick_man.4 #sick_man.2
		}
		#kmpmp
		add_loyalists = {
			value = 0.07
		}
		add_modifier = {
			name = kmpmp_sick_man_of_europe_restored_fame
		}
	}

	timeout = 10950 #kmpmp 7300

	on_timeout = {
		remove_modifier = sick_man_of_europe
		add_modifier = {
			name = dead_man_of_europe
			days = very_long_modifier_time
			is_decaying = yes
		}
		trigger_event = {
			id = sick_man.3
		}
		if = {
			limit = {
				is_in_power_bloc = yes
				is_power_bloc_leader = yes				
			}
			custom_tooltip = sick_man_of_europe_lose_power_bloc
		}
		custom_tooltip = {
			text = je_sick_man_main_failure_movements_tt
			set_variable = {
				name = failed_tanzimat
				years = 20
			}
		}
		if = {
			limit = {
				has_dlc_feature = ip3_content
			}
			custom_tooltip = {
				text = eastern_crisis_consequence_tt
				set_variable = failed_sick_man_main
			}
		}
		if = { #failure immediately triggers eastern_crisis
			limit = {
				c:TUR ?= this
				has_dlc_feature = ip3_content
				NOT = {
					has_variable = eastern_crisis_happened
				}
			}
			add_journal_entry = {
				type = je_great_eastern_crisis
			}
		}
		cu:albanian = {
			remove_tradition_trait = tradition_rumelian
		}
		cu:bosniak = {
			remove_tradition_trait = tradition_rumelian
		}
		cu:sephardic = {
			remove_tradition_trait = tradition_rumelian
		}
	}

	current_value = {
		value = var:sick_man_var
	}

	goal_add_value = {
		add = 4
	}

	progressbar = yes

	weight = 10000

	should_be_pinned_by_default = yes
}

# Literacy and Universities
je_sick_man_education = {
	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"

	group = je_group_tanzimat

	complete = {
		any_scope_building = {
			is_building_type = building_university
			occupancy > 0.9
			level >= 5
		}
		scope:journal_entry = {
			is_goal_complete = yes
		}
	}

	on_complete = {
		set_variable = completed_sick_man_education_je
		custom_tooltip = je_sick_man_advance
		change_variable = {
			name = sick_man_var
			add = 1
		}
	}

	current_value = {
		value = root.literacy_rate
		multiply = 100 #kmpmp for better progress optics
	}

	goal_add_value = {
		add = 0.2
		multiply = 100 #kmpmp for better progress optics
	}

	invalid = {
		custom_tooltip = {
			text = finished_sick_man_tt
			has_variable = sick_man_complete
		}
	}

	transferable = no

	weight = 5000

	should_be_pinned_by_default = yes

	progressbar = yes
}
