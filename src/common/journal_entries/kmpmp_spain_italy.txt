            }

            limit = { NOT = { has_claim = s:STATE_APULIA } }
            s:STATE_APULIA = {
                add_claim = ROOT
            }
        }
        if = {
            limit = { NOT = { has_claim = s:STATE_CALABRIA } }
            s:STATE_CALABRIA = {
                add_claim = ROOT
            }
        }
