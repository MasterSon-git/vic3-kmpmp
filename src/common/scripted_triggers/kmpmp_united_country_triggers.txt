kmpmp_owns_all_ibe_states = {
    trigger_if = {
        limit = {
            exists = global_var:states_ibe
        }
        any_in_global_list = {
            variable = states_ibe
            any_scope_state = {
                owner = ROOT
                count = all
            }
            count = all
        }
    }
    trigger_else = {
        always = no
    }
}
