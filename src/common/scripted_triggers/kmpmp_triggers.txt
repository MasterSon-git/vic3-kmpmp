# root - country
# STATE_REGION - <s:STATE_***>
owns_or_has_subject_entire_state_region = {
    $STATE_REGION$ = {
        custom_tooltip = {
            text = owns_or_has_subject_entire_state_region_tooltip
            any_scope_state = {
                OR = {
                    owner = root
                    owner = {
                        is_subject_of = root
                    }
                }
                count = all
            }
        }
    }
}

state_region_owned_and_incorporated = {

        text = state_region_owned_and_incorporated_tooltip

        }
    }
}
