kmpmp_fix_migrated_non_accepted_pops = {
    every_scope_state = {
        limit = {
            owner = ROOT
            # is_incorporated = yes
            any_scope_culture = {
                has_homeland = PREV
                NOT = {
                    culture_is_discriminated_in = ROOT
                }
            }
        }

        save_temporary_scope_as = current_state
        # get culture with most accepted pops in state
        ordered_scope_culture = {
            limit = {
                has_homeland = scope:current_state
                NOT = {
                    culture_is_discriminated_in = ROOT
                }
            }
            order_by = {
                save_temporary_scope_as = culture_to_check
                value = 0
                scope:current_state = {
                    if = {
                        limit = {
                            culture_percent_state = { target = scope:culture_to_check value > 0.1 }
                        }
                        add = 1
                    }
                    if = {
                        limit = {
                            culture_percent_state = { target = scope:culture_to_check value > 0.2 }
                        }
                        add = 1
                    }
                    if = {
                        limit = {
                            culture_percent_state = { target = scope:culture_to_check value > 0.3 }
                        }
                        add = 1
                    }
                    if = {
                        limit = {
                            culture_percent_state = { target = scope:culture_to_check value > 0.4 }
                        }
                        add = 1
                    }
                    if = {
                        limit = {
                            culture_percent_state = { target = scope:culture_to_check value > 0.5 }
                        }
                        add = 1
                    }
                    if = {
                        limit = {
                            culture_percent_state = { target = scope:culture_to_check value > 0.6 }
                        }
                        add = 1
                    }
                    if = {
                        limit = {
                            culture_percent_state = { target = scope:culture_to_check value > 0.7 }
                        }
                        add = 1
                    }
                    if = {
                        limit = {
                            culture_percent_state = { target = scope:culture_to_check value > 0.8 }
                        }
                        add = 1
                    }
                    if = {
                        limit = {
                            culture_percent_state = { target = scope:culture_to_check value > 0.9 }
                        }
                        add = 1
                    }
                }
            }
            save_temporary_scope_as = target_culture
        }
        # convert non accepted pops to target_culture
        every_scope_pop = {
            limit = {
                NOT = {
                    culture = {
                        has_homeland = scope:current_state
                    }
                }
                pop_is_discriminated = yes
                culture_accepted = no
            }
            kmpmp_change_pop_culture = yes
