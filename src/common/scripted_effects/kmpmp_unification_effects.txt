kmpmp_arabia_unification_modifier = {
	if = {
		limit = {
			has_modifier = kmpmp_ottoman_leader_of_islam
		}
		add_modifier = {
			name = kmpmp_united_arabia
			years = 10
			multiplier = 0.5
		}
	}
	else = {
		add_modifier = {
			name = kmpmp_united_arabia
			years = 10
		}
		add_modifier = {
			name = kmpmp_arabia
		}
	}
}

kmpmp_arabia_unification_effect = {
	# claims
	every_state = {
		limit = {
			NOT = { owner = ROOT }
			OR = {
				state_region = s:STATE_KONYA
				state_region = s:STATE_HUDAVENDIGAR
				state_region = s:STATE_AYDIN
				state_region = s:STATE_KASTAMONU
				state_region = s:STATE_ANKARA
			}
		}
		
		state_region = { add_claim = ROOT }
	}
}

kmpmp_unification_claims_effect_ltn = {
	if = {
		limit = {
			any_in_global_list = {
				variable = states_ltn

				any_scope_state = {
					# is_homeland_of_country_cultures = ROOT
					NOT = { owner = ROOT }
					NOT = { has_claim_by = ROOT }
				}
			}
		}

		every_in_global_list = {
			variable = states_ltn
	
			if = {
				limit = {
					any_scope_state = {
						# is_homeland_of_country_cultures = ROOT
						NOT = { owner = ROOT }
						NOT = { has_claim_by = ROOT }
					}
				}
				add_claim = ROOT
			}
		}
	}
	else = {
		add_loyalists = {
			value = 0.05
		}
	}
}
